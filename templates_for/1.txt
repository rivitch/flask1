from flask import Flask

from flask import request # даёт доступ к локальной информации для каждого контекста запроса. содержит данные, 
                          # которую клиент передаёт на сторону сервера.
from flask import url_for                          

app = Flask(__name__)


@app.route('/')
def index():
    return 'Введи путь к файлу в адресной строке'


@app.route('/<path:file>/')
def get_file(file):
    print(file)
    # return f'Ваш файл находится в: {file}!'  # без экранирования
    return f'Ваш файл находится в: {escape(file)}!'  # с экранированием

if __name__ == "__main__":
